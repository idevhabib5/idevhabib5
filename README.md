    animation: slideIn 1s ease-out forwards;

    animation: slideIn 2s ease-out forwards;

  @keyframes slideIn {
    from {
      transform: translateX(-100%);
      opacity: 0;
    }
    to {
      transform: translateX(0);
      opacity: 1;
    }
  }
</style>
